package sam;
import java.sql.Connection;
import java.sql.DriverManager;
import java.sql.Statement;

public class Sql {
    public static void main(String[] args) {
        try {
            Connection con = DriverManager.getConnection("jdbc:mysql://localhost", "root", "root");
            Statement stat = con.createStatement();
            stat.execute("CREATE DATABASE IF NOT EXISTS dem");
            con = DriverManager.getConnection("jdbc:mysql://localhost/dem", "root", "root");
            stat = con.createStatement();
            stat.execute("CREATE TABLE IF NOT EXISTS stud (id INT PRIMARY KEY AUTO_INCREMENT, name VARCHAR(100), AGE INT)");
            System.out.println("Database and table created");
            con.close();
        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}
